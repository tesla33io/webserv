
location = / {
	[ configuration A ]
}

location / {
	[ configuration B ]
}

location /documents/ {
	[ configuration C ]
}

location ^~ /images/ {
	[ configuration D ]
}

location ~* \.(gif|jpg|jpeg)$ {
	[ configuration E ]
}

server {
	listen 8080;
	server_name localhost www.local.dev;

	root ./www;
	index index.html index.htm;

	autoindex off;

	error_page 404 /errors/404.html;
	error_page 500 502 503 504 /errors/50x.html;

	client_max_body_size 1M;

	location / {
		try_files $uri $uri/ =404;
	}

	location /images/ {
		root ./static/img;
		autoindex on;
	}

	location /css/ {
		root ./static/css;
	}

	location /upload/ {
		root ./uploads;
		client_max_body_size 10M;
	}

	location /api/ {
		return 403;
	}

	location /redirect-me {
		return 301 http://example.com;
	}

	location ~ \.php$ {
		fastcgi_pass 127.0.0.1:9000;
		fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
		include fastcgi_params;
	}

	location /errors/ {
		root ./errors;
	}

	location /download/ {
		root ./downloads;
		autoindex on;
	}
}
