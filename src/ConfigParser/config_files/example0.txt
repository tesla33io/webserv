	server {

		host 127.0.0.1;

		}

		location / {

		server_name second;
		host 127.0.0.1;
		index       index.html;

		location /images/ {
			root ./img;
			allow_methods GET POST;
		}

		location / {
			autoindex on;
		}
	}
