
http {
	server {
	
		listen 127.0.0.1:8080;
		server_name mysite.local www.mysite.local;
		# client_max_body_size 10M;

		location / {
			root /var/www/html;
			index index.html;
			autoindex off;
			limit_except GET POST;  # no DEL
			error_page 404 /custom_404.html;
			error_page 500 502 503 504 /custom_50x.html;
		}

		location /uploads {
			root /var/www/uploads;
			limit_except POST GET;
			autoindex on;
		}

		location /redirect-me {
			return 301 http://example.com/;
		}

		location /cgi-bin {
			root /var/www/cgi-bin;
			limit_except GET POST;
			cgi_ext .py;
			# All *.py files in this location are executed as CGI
		}

		location /images {
			root /var/www/images;
			autoindex on;
			index index.jpg;
			limit_except GET;
		}   
	}
}
