

server {
    listen 8080;
    host 127.0.0.1;
    server_name mysite.local www.mysite.local;

    error_page 404 /custom_404.html;
    error_page 500 502 503 504 /custom_50x.html;
    client_max_body_size 10M;

    location / {
        root /var/www/html;
        index index.html;
        autoindex off;
        allow_methods GET POST;  # no DEL
    }

    location /uploads {
        root /var/www/uploads;
        allow_methods POST GET;
        # Custom directive for upload directory (if implemented)
        upload_store /var/www/uploads;
        autoindex on;
    }

    location /redirect-me {
        return 301 http://example.com/;
    }

    location /cgi-bin {
        root /var/www/cgi-bin;
        allow_methods GET POST;
        cgi_ext .py;
        cgi_path /usr/bin/python3;
        # All *.py files in this location are executed as CGI
    }

    location /images {
        root /var/www/images;
        autoindex on;
        index index.jpg;
        allow_methods GET;
    }
}
